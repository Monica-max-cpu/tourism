### 4.4 新建:`src/api/common/dict.ts`

```
/**
 * JeecgBoot 字典接口封装
 * update-begin--author:phase7---date:2026-05-25---for:【阶段7】对接 sys/dict/getDictItems
 * - 客户端 Map 缓存,同一 dictCode 仅请求一次
 * - 失败时降级为空数组,不阻塞业务渲染
 * update-end--author:phase7---date:2026-05-25---for:【阶段7】对接 sys/dict/getDictItems
 */
import { defHttp } from '/@/api/http';

export interface DictItem {
  label: string;
  value: string;
}

const cache = new Map<string, DictItem[]>();

export async function getDictItems(dictCode: string): Promise<DictItem[]> {
  if (cache.has(dictCode)) return cache.get(dictCode)!;
  try {
    const list = await defHttp.get<DictItem[]>({ url: `/sys/dict/getDictItems/${dictCode}` });
    const safe = Array.isArray(list) ? list : [];
    cache.set(dictCode, safe);
    return safe;
  } catch {
    return [];
  }
}

export function clearDictCache(code?: string) {
  if (code) cache.delete(code);
  else cache.clear();
}
```

### 5.2 新建:`src/views/login/CasCallback.vue`

```
<!--
  CAS 回调中转页:URL 形如 /cas/callback?ticket=ST-xxx&redirect=/xxx
  update-begin--author:phase7---date:2026-05-25---for:【阶段7】CAS 单点
  update-end--author:phase7---date:2026-05-25---for:【阶段7】CAS 单点

<template>
  <div class="flex h-screen items-center justify-center text-muted-foreground">
    {{ message }}
  </div>
</template>

<script setup lang="ts">
import { ref, onMounted } from 'vue';
import { useRoute, useRouter } from 'vue-router';
import { useUserStore } from '/@/stores/modules/user';
import { ROUTE_PATHS } from '/@/constants/routePaths';

const route = useRoute();
const router = useRouter();
const userStore = useUserStore();
const message = ref('正在校验 CAS 票据...');

const ROLE_HOME: Record<string, string> = {
  ADMIN: ROUTE_PATHS.ADMIN_WORKBENCH,
  SUPPLIER: ROUTE_PATHS.SUPPLIER_WORKBENCH,
  STORE: ROUTE_PATHS.STORE_WORKBENCH,
};

onMounted(async () => {
  const ticket = route.query.ticket as string;
  if (!ticket) {
    message.value = '缺少 ticket 参数,2 秒后回登录页';
    setTimeout(() => router.replace(ROUTE_PATHS.LOGIN), 2000);
    return;
  }
  try {
    const user = await userStore.loginByCAS(ticket);
    const redirect = (route.query.redirect as string) || ROLE_HOME[user.role] || ROUTE_PATHS.LOGIN;
    router.replace(redirect);
  } catch (e: any) {
    message.value = `CAS 校验失败:${e?.message || '未知错误'},2 秒后回登录页`;
    setTimeout(() => router.replace(ROUTE_PATHS.LOGIN), 2000);
  }
});
</script>
```

## 联调对接

- **后端**:JeecgBoot,默认 `/jeecgboot` 代理到 `http://localhost:8080`
- **登录**:`POST /sys/login` → `{ token, userInfo }`,token 存 `localStorage('b2b:token')`,header `X-Access-Token`
- **字典**:`GET /sys/dict/getDictItems/{code}`,客户端缓存
- **CAS**:登录页 → `${VITE_CAS_SERVER}/login?service=${origin}/cas/callback` → 回调换 token
- **防重放**:每个请求自动带 `X-Timestamp` + `X-Nonce`
- **多租户**:可选 `X-Tenant-Id`(从 `localStorage('b2b:tenantId')`)
- **qiankun**:子应用名 `b2b-procurement`,base `/b2b-procurement/`

## 12. 附录:AI 执行者元规则

> 这一章对纯 AI 执行者(下一棒)有约束力。

### 12.1 调用工具的强制顺序

每个子任务,AI 执行者必须按以下顺序操作:

1. **read** 所有 `read_first` 列出的文件(完整,不要 offset)
2. 校验文件内容是否与本文档「old_string」匹配(逐字符比对)
3. 不匹配 → **停止**,在对话中报告差异,等用户确认,**禁止** 自行猜测
4. 匹配 → 执行 `edit` / `write` / `apply_patch`
5. 跑「verify.cmd」,失败 → 报告原始错误,**禁止** 静默重试 3 次以上
6. 通过 → 进入下一个子任务

### 12.2 改动幂等性

所有改动必须**可重跑**:

- 重跑同一子任务,二次执行应不产生重复 `update-begin/end` 块
- 如发现已存在同名 marker,跳过该改动并报告「已应用」
- 不要写时间戳到 marker(date 字段固定 `2026-05-25`)

### 12.3 错误回滚

如果某一子任务 verify 失败:

1. 报告失败原因(原始错误,不要总结)
2. 不要主动改其他文件「修复」
3. 用户决定:回滚 / 修补 / 跳过

回滚命令:

```
git checkout HEAD -- <file>          # 单文件
git reset --hard HEAD                # 整次提交
git checkout main                    # 切回主分支放弃整个分支
```
